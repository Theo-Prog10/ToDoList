using Microsoft.AspNetCore.Mvc;
using Model.requests;

namespace Controllers;

[ApiController]
[Route("api/[controller]")]
public class PessoaController : ControllerBase
{
    private readonly GetAllPessoasHandler _getAllHandler;
    private readonly GetPessoaByIdHandler _getByIdHandler;

    public PessoaController(GetAllPessoasHandler getAllHandler, GetPessoaByIdHandler getByIdHandler)
    {
        _getAllHandler = getAllHandler;
        _getByIdHandler = getByIdHandler;
    }

    [HttpGet]
    public async Task<IActionResult> GetAll()
    {
        var pessoas = await _getAllHandler.Handle();
        return Ok(pessoas);
    }

    [HttpGet("{id}")]
    public async Task<IActionResult> GetById(int id)
    {
        var pessoa = await _getByIdHandler.Handle(id);
        if (pessoa == null)
        {
            return NotFound();
        }
        return Ok(pessoa);
    }

    
}
